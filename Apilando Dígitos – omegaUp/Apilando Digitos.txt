#include <stdio.h>

void mostrarPiramide(int n)
{
    for (int i = 1; i <= n; i++)
    {
        for (int j = 0; j < i; j++)
        {
            printf("%d", i);
        }
        printf("\n");
    }
    for (int i = n - 1; i >= 1; i--)
    {
        for (int j = 0; j < i; j++)
        {
            printf("%d", i);
        }
        printf("\n");
    }

}

int main() {
	
	int N;
	scanf("%d", &N);
	
	mostrarPiramide(N);
	
	return 0;
}
